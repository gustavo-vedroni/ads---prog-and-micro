    <h1>Exemplo 04</h1>
    <p>Calcula o valor com desconto a partir do preço e do percentual de desconto.</p>

    <form id="form">
        <input id="field1" type="number" placeholder="Digite o primeiro valor">
        <button>Calcular</button>
    </form>

    <div id="result"></div>

    <script>

        function onSubmit(event) {

            event.preventDefault()

            let num = parseFloat(field1.value)        
            
            
            
            for (let i = 1; i <= 10; i++)
            {
                let resultado = num * i;
                result.innerHTML += `<li>${i} × ${num} = <strong>${resultado}</strong></li>`
            }
            
            
            
        }
        form.addEventListener('submit', onSubmit)
    </script>
